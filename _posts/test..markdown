function readCSVFile() {
    var data = [];


    // read data from second column of test.csv
    var csv = [];
    
    $.ajax({
        type: "GET",
        url: "test.csv",
        dataType: "text",
        success: function(response)
        {
            csv = $.csv.toArrays(response);
            for (var i = 0; i < csv.length; i++) {
                data[i] = parseFloat(csv[i][1]);
            }
        }
     });



    return data;
}
